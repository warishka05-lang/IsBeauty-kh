import { Router } from 'express'; import { getContent, saveContent } from '../services/content.service.js'; import { requireAuth } from '../middleware/auth.middleware.js'; const router = Router(); router.get('/', (_req, res) => { res.json(getContent()); }); router.put('/', requireAuth, (req, res) => { const payload = req.body; if (!payload || typeof payload !== 'object') { return res.status(400).json({ message: 'Некорректные данные' }); } return res.json(saveContent(payload)); }); export default router;
