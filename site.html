const express = require('express'); const bcrypt = require('bcryptjs'); const jwt = require('jsonwebtoken'); const prisma = require('../lib/prisma'); const router = express.Router(); router.post('/login', async (req, res) => { const { login, password } = req.body || {}; const user = await prisma.user.findUnique({ where: { login } }); if (!user) return res.status(401).json({ message: 'Invalid credentials' }); const ok = bcrypt.compareSync(password || '', user.passwordHash); if (!ok) return res.status(401).json({ message: 'Invalid credentials' }); const token = jwt.sign({ id: user.id, login: user.login, role: user.role }, process.env.JWT_SECRET, { expiresIn: '12h' }); res.json({ token, user: { login: user.login, role: user.role } }); }); module.exports = router;
