const express = require('express'); const prisma = require('../lib/prisma'); const auth = require('../middleware/auth'); const defaultContent = require('../lib/defaultContent'); const router = express.Router(); router.get('/content', async (_, res) => { let row = await prisma.siteContent.findUnique({ where: { id: 'main' } }); if (!row) row = await prisma.siteContent.create({ data: { id: 'main', payload: defaultContent } }); res.json(row.payload); }); router.put('/content', auth, async (req, res) => { const row = await prisma.siteContent.upsert({ where: { id: 'main' }, update: { payload: req.body || defaultContent }, create: { id: 'main', payload: req.body || defaultContent } }); res.json({ success: true, content: row.payload }); }); module.exports = router;
